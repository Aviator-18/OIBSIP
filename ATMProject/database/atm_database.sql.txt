-- Create Database and use it
CREATE DATABASE IF NOT EXISTS atm_system;
USE atm_system;

-- Drop helper tables/procedures if exist (safe to re-run)
DROP TABLE IF EXISTS first_names;
DROP TABLE IF EXISTS last_names;
DROP PROCEDURE IF EXISTS populate_users;
DROP PROCEDURE IF EXISTS populate_transactions;

-- Users table
CREATE TABLE IF NOT EXISTS users (
    user_id VARCHAR(20) PRIMARY KEY,
    pin VARCHAR(4) NOT NULL,
    account_number VARCHAR(20) UNIQUE NOT NULL,
    account_holder VARCHAR(100) NOT NULL,
    balance DECIMAL(15, 2) NOT NULL DEFAULT 0.00,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Transactions table
CREATE TABLE IF NOT EXISTS transactions (
    transaction_id INT AUTO_INCREMENT PRIMARY KEY,
    account_number VARCHAR(20) NOT NULL,
    transaction_type VARCHAR(20) NOT NULL,
    amount DECIMAL(15, 2) NOT NULL,
    recipient_account VARCHAR(20),
    transaction_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    description VARCHAR(255),
    FOREIGN KEY (account_number) REFERENCES users(account_number)
);

-- Helper name lists
CREATE TABLE first_names (name VARCHAR(50) PRIMARY KEY);
CREATE TABLE last_names  (name VARCHAR(50) PRIMARY KEY);

-- Procedure to populate 500 users with random realistic names, pins and balances
DELIMITER //
CREATE PROCEDURE populate_users(IN howmany INT)
BEGIN
    DECLARE i INT DEFAULT 1;
    DECLARE fname VARCHAR(50);
    DECLARE lname VARCHAR(50);
    DECLARE uname VARCHAR(100);
    DECLARE uid VARCHAR(20);
    DECLARE acc VARCHAR(20);
    DECLARE pin4 VARCHAR(4);
    DECLARE bal DECIMAL(15,2);

    START TRANSACTION;
    WHILE i <= howmany DO
        -- pick a random first and last name
        SELECT name INTO fname FROM first_names ORDER BY RAND() LIMIT 1;
        SELECT name INTO lname FROM last_names ORDER BY RAND() LIMIT 1;
        SET uname = CONCAT(fname, ' ', lname);

        -- user id and account number pattern: user001..user500 and ACC001..ACC500
        SET uid = CONCAT('user', LPAD(i, 3, '0'));
        SET acc = CONCAT('ACC', LPAD(i, 3, '0'));

        -- random 4-digit PIN (zero-padded)
        SET pin4 = LPAD(FLOOR(RAND() * 10000), 4, '0');

        -- realistic random balance: between 500.00 and 100000.00 (rounded to 2 decimals)
        SET bal = ROUND(500 + (RAND() * 99500), 2);

        INSERT INTO users (user_id, pin, account_number, account_holder, balance)
        VALUES (uid, pin4, acc, uname, bal);

        SET i = i + 1;
    END WHILE;
    COMMIT;
END //
DELIMITER ;

-- Create 500 users
CALL populate_users(500);

-- Procedure to generate random transactions and update balances accordingly
-- params: howmany = # of transactions to create
DELIMITER //
CREATE PROCEDURE populate_transactions(IN howmany INT)
BEGIN
    DECLARE j INT DEFAULT 1;
    DECLARE acc_from VARCHAR(20);
    DECLARE acc_to VARCHAR(20);
    DECLARE ttype INT;
    DECLARE amt DECIMAL(15,2);
    DECLARE descr VARCHAR(255);
    DECLARE from_balance DECIMAL(15,2);

    START TRANSACTION;
    WHILE j <= howmany DO
        -- pick random accounts for from and to
        SELECT account_number INTO acc_from FROM users ORDER BY RAND() LIMIT 1;
        SELECT account_number INTO acc_to   FROM users ORDER BY RAND() LIMIT 1;

        -- ensure from != to for transfer operations; if equal, pick another to
        IF acc_from = acc_to THEN
            SELECT account_number INTO acc_to FROM users WHERE account_number <> acc_from ORDER BY RAND() LIMIT 1;
        END IF;

        -- choose transaction type randomly
        -- 1 => DEPOSIT, 2 => WITHDRAWAL, 3 => TRANSFER
        SET ttype = FLOOR(1 + RAND() * 3);

        -- random amount: small to medium amounts, 20.00 - 20000.00
        SET amt = ROUND(20 + (RAND() * 19980), 2);

        IF ttype = 1 THEN
            -- DEPOSIT into acc_from
            SET descr = CONCAT('Deposit of ', amt);
            INSERT INTO transactions (account_number, transaction_type, amount, recipient_account, description)
            VALUES (acc_from, 'DEPOSIT', amt, NULL, descr);
            -- update balance
            UPDATE users SET balance = balance + amt WHERE account_number = acc_from;

        ELSEIF ttype = 2 THEN
            -- WITHDRAWAL from acc_from, ensure sufficient funds; if not, skip or reduce amount
            SELECT balance INTO from_balance FROM users WHERE account_number = acc_from FOR UPDATE;
            IF from_balance < 20 THEN
                -- too low to withdraw; skip this iteration by inserting a small deposit instead
                SET amt = ROUND(50 + (RAND() * 450), 2);
                SET descr = CONCAT('Auto-deposit fallback ', amt);
                INSERT INTO transactions (account_number, transaction_type, amount, recipient_account, description)
                VALUES (acc_from, 'DEPOSIT', amt, NULL, descr);
                UPDATE users SET balance = balance + amt WHERE account_number = acc_from;
            ELSE
                IF amt > from_balance THEN
                    SET amt = ROUND(from_balance * RAND(), 2);
                    IF amt < 20 THEN SET amt = 20.00; END IF;
                END IF;
                SET descr = CONCAT('Withdrawal of ', amt);
                INSERT INTO transactions (account_number, transaction_type, amount, recipient_account, description)
                VALUES (acc_from, 'WITHDRAWAL', amt, NULL, descr);
                UPDATE users SET balance = balance - amt WHERE account_number = acc_from;
            END IF;

        ELSE
            -- TRANSFER from acc_from -> acc_to
            -- ensure acc_from has enough balance; if not, adjust amount or convert to deposit
            SELECT balance INTO from_balance FROM users WHERE account_number = acc_from FOR UPDATE;
            IF from_balance < 20 THEN
                -- not enough to transfer; do a deposit to acc_from instead
                SET amt = ROUND(50 + (RAND() * 450), 2);
                SET descr = CONCAT('Auto-deposit fallback ', amt);
                INSERT INTO transactions (account_number, transaction_type, amount, recipient_account, description)
                VALUES (acc_from, 'DEPOSIT', amt, NULL, descr);
                UPDATE users SET balance = balance + amt WHERE account_number = acc_from;
            ELSE
                IF amt > from_balance THEN
                    SET amt = ROUND(from_balance * RAND(), 2);
                    IF amt < 10 THEN SET amt = 10.00; END IF;
                END IF;
                SET descr = CONCAT('Transfer to ', acc_to);
                INSERT INTO transactions (account_number, transaction_type, amount, recipient_account, description)
                VALUES (acc_from, 'TRANSFER', amt, acc_to, descr);
                -- subtract from sender and add to recipient
                UPDATE users SET balance = balance - amt WHERE account_number = acc_from;
                UPDATE users SET balance = balance + amt WHERE account_number = acc_to;
            END IF;
        END IF;

        SET j = j + 1;
    END WHILE;
    COMMIT;
END //
DELIMITER ;

-- Generate between 50 and 80 transactions (choose 70 here)
CALL populate_transactions(70);
